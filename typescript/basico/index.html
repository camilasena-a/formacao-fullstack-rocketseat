// Definindo tipos
type StatusTarefa = 'pendente' | 'em-progresso' | 'concluida';

interface Tarefa {
    id: number;
    titulo: string;
    descricao?: string;
    status: StatusTarefa;
    criadaEm: Date;
    concluidaEm?: Date;
}

// Classe para gerenciar tarefas
class GerenciadorTarefas {
    private tarefas: Tarefa[] = [];
    private proximoId: number = 1;

    adicionarTarefa(titulo: string, descricao?: string): Tarefa {
        const novaTarefa: Tarefa = {
            id: this.proximoId++,
            titulo,
            descricao,
            status: 'pendente',
            criadaEm: new Date()
        };

        this.tarefas.push(novaTarefa);
        return novaTarefa;
    }

    alterarStatus(id: number, novoStatus: StatusTarefa): boolean {
        const tarefa = this.tarefas.find(t => t.id === id);
        
        if (!tarefa) return false;

        tarefa.status = novoStatus;
        
        if (novoStatus === 'concluida') {
            tarefa.concluidaEm = new Date();
        }

        return true;
    }

    listarTarefas(status?: StatusTarefa): Tarefa[] {
        if (status) {
            return this.tarefas.filter(t => t.status === status);
        }
        return [...this.tarefas];
    }

    removerTarefa(id: number): boolean {
        const indice = this.tarefas.findIndex(t => t.id === id);
        
        if (indice === -1) return false;
        
        this.tarefas.splice(indice, 1);
        return true;
    }
}
